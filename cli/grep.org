* 显示匹配内容在文件中的偏移量

#+BEGIN_SRC sh
  grep -b "text" <filename>
#+END_SRC


* 匹配多个式样

#+BEGIN_SRC sh
  grep -e "text1" -e "text2"
#+END_SRC


* 指定文件类型搜索

#+BEGIN_SRC sh
  # 搜索所有 c 和 cpp 文件
  grep -R "text" . --include *.{c,cpp}
#+END_SRC

* 排除文件

#+BEGIN_SRC sh
  grep -R "text" . --exclude "README"
#+END_SRC

* 排除目录

#+BEGIN_SRC sh
  grep -R "text" . --exclude-dir=.git
#+END_SRC


* 忽略二进制文件

#+BEGIN_SRC sh
  grep -I
#+END_SRC
