* 打包目录并传至远端服务器

  #+BEGIN_SRC sh
    tar -c <dir> | gzip | ssh user@remote 'dd of=dir.tar.gz'
    tar -c <dir> | ssh -C user@remote 'cd /path && tar -x -p'
  #+END_SRC

* 打包离散文件

  #+BEGIN_SRC sh
    find . -name '*.txt' | tar -c --files-from=- | bzip2 > dir_txt.tar.bz2
  #+END_SRC
