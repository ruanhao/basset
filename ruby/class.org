* Singleton
#+BEGIN_SRC ruby
## Singleton (not thread safe)

class MyLogger
  private_class_method :new
  @@logger = nil
  def MyLogger.create
    @@logger = new unless @@logger
    @@logger
  end
end
#+END_SRC

* Class name
If a non-name class is assigned to a constant, this constant is treated as the name of the class
#+BEGIN_SRC ruby
  var = Class.new
  var.name ## nil

  Wibble = var
  var.name ## "Wibble"
#+END_SRC
