(arr ||= []) << "value" ## 'arr ||= []' is 'arr = arr || []'

%W(#{foo} Bar Bar\ with\ space) ## like "
## => ["Foo", "Bar", "Bar with space"] )

%w(a b c\ d \#e #{1}f) ## like '
## => ["a", "b", "c d", "\\#e", "\#{1}f"])

%i(a b c)
## => [:a, :b, :c]
