* Array
** Assign
#+BEGIN_SRC ruby
a     = [1, 3, 5, 7, 9]    # -> [1, 3, 5, 7, 9]
a[1]  = 'bat'              # -> [1, "bat", 5, 7, 9]
a[-3] = 'cat'              # -> [1, "bat", "cat", 7, 9]
a[3]  = [9, 8]             # -> [1, "bat", "cat", [9, 8], 9]
a[6]  = 99                 # -> [1, "bat", "cat", [9, 8], 9, nil, 99]
#+END_SRC

#+BEGIN_SRC ruby
a       = [1, 3, 5, 7, 9]  # -> [1, 3, 5, 7, 9]
a[2, 2] = 'cat'            # -> [1, 3, "cat", 9]
a[2, 0] = 'dog'            # -> [1, 3, "dog", "cat", 9]
a[1, 1] = [9, 8, 7]        # -> [1, 9, 8, 7, "dog", "cat", 9]
a[0..3] = []               # -> ["dog", "cat", 9]
a[5..6] = 99, 98           # -> ["dog", "cat", 9, nil, nil, 99, 98]
#+END_SRC
