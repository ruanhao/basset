- 文件并集，交集，差集

  #+BEGIN_SRC sh
    cat a b | sort | uniq > c        # c is a union b 并集
    cat a b | sort | uniq -d > c     # c is a intersect b 交集
    cat a b b | sort | uniq -u > c   # c is set difference a - b 差集
  #+END_SRC
