** DispatcherServlet

In web.xml

#+BEGIN_SRC xml
  <servlet>
    <servlet-name>xxx</servlet-name>
    <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
    <load-on-startup>1</load-on-startup>
  </servlet>

  <servlet-mapping>
    <servlet-name>xxx</servlet-name>
    <url-pattern>*.html</url-pattern>
  </servlet-mapping>
#+END_SRC
